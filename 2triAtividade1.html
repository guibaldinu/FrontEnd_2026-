const textInput = document.getElementById('taskInput'); const taskList = document.getElementById('taskList'); function addTask(){ const textInput = taskInput.value.trim(); if (textInput !== ''){ const newItem = document.createElement('li'); newItem.innerHTML = `${textInput} ` taskList.appendChild(newItem); } } function remover(button){ const itemToRemove = button.parentElement; taskList.removeChild(itemToRemove) } function concluir(button){ const itemToComplete = button.parentElement; itemToComplete.classList.toggle(`completed`) }
